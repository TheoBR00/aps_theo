COMPARISONS-TERM = * | / | &&

FACTOR = NUMBER | STR | IDENT, IGUAL-OP, RELEXP-FUNC | +, FACTOR | -, FACTOR | !, FACTOR | (, RELEXP-FUNC, ) | LEITURA

NÚMERO = ALGARISMO

IDENT = LETRA {LETRA, NÚMERO, _}

VARDEC = IDENT | {, IDENT, }, :, TIPO

SE = (, RELEXP-FUNC, ), STATEMENT-FUNC

LEITURA = ()
